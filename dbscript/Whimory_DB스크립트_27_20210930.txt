/* 회원정보 */
CREATE TABLE TB_USER (
	USER_ID VARCHAR2(30) NOT NULL, /* 회원 아이디 */
	USER_PWD VARCHAR2(100) NOT NULL, /* 비밀번호 */
	USER_NAME VARCHAR2(20) NOT NULL, /* 회원 이름 */
	GENDER CHAR(1) NOT NULL, /* 성별 */
	BIRTH DATE NOT NULL, /* 생년월일 */
	PHONE VARCHAR2(20) NOT NULL, /* 전화번호 */
	EMAIL VARCHAR2(50) NOT NULL, /* 이메일 */
	AUTH_STATUS CHAR(1) DEFAULT 'N', /* 인증여부 */
	JOIN_DATE DATE DEFAULT SYSDATE, /* 가입일 */
	LAST_LOGIN_DATE DATE DEFAULT SYSDATE, /* 마지막 접속일 */
	ADMIN_YN CHAR(1) DEFAULT 'N', /* 관리자 여부 */
	LOGIN_ACCESS_YN CHAR(1) DEFAULT 'Y', /* 로그인 가능 여부 */
	PENALTY_COUNT NUMBER DEFAULT 0); /* 경고 횟수 */

COMMENT ON COLUMN TB_USER.USER_NAME IS '회원 이름';

COMMENT ON COLUMN TB_USER.GENDER IS '성별';

COMMENT ON COLUMN TB_USER.BIRTH IS '생년월일';

COMMENT ON COLUMN TB_USER.PHONE IS '전화번호';

COMMENT ON COLUMN TB_USER.EMAIL IS '이메일';

COMMENT ON COLUMN TB_USER.AUTH_STATUS IS '인증여부';

COMMENT ON COLUMN TB_USER.JOIN_DATE IS '가입일';

COMMENT ON COLUMN TB_USER.LAST_LOGIN_DATE IS '마지막 접속일';

COMMENT ON COLUMN TB_USER.ADMIN_YN IS '관리자 여부';

COMMENT ON COLUMN TB_USER.LOGIN_ACCESS_YN IS '로그인 가능 여부';
